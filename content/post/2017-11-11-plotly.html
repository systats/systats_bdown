  filter(year %in% 2000) %>% #
  rename(value = exports) %>%
  select(region, country, value) %>%
  mutate(region = stringr::str_replace_all(region, "\\(.*?\\)", "")) %>%
  mutate(region = stringr::str_replace_all(region, "-", " ")) %>%
  arrange(region, value) %>%
  mutate(seqs = paste(region, country, sep = "-")) %>%
  select(seqs, value) %>%
  sunburst(count = T, percent = T)
